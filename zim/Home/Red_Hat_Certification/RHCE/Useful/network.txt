**Port checking**
ss -tlnp  | grep mysql  #see if sql is listening on ports.
		 #t=tcp.  #l=listening  #n=numeric ports.  #p=ports

nmap -p 22 HOST   # test if a port is open. (needs to be installed first).
cat < /dev/tcp/localhost/22 # reads port.

telnet HOST PORT  # connect to a port!
