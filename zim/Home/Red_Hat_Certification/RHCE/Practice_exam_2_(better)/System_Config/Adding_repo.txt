cd /etc/yum.repos.d/
vim rhel_dvd.repo
[rhel_dvd]
gpgcheck = 0
enabled = 1
baseurl = https://content.example.com/rhel7.0/x86_64/dvd
name = Remote classroom copy of dvd

yum clean all
yum install "*samba*"  # or some pkg, maybe nmap.
