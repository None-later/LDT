====== MariaDB ======
Created Sunday 18 March 2018

wget  http://classroom.example.com/materials/mariadb/mariadb.dump

yum -y groupinstall mariadb*
vim /etc/my.conf
| 	skip-networking=1

systemctl restart mariadb
systemctl enable mariadb
firewall-cmd --permanent --add-service=mysql
firewall-cmd --reload

/> create database legacy ;
/> exit

mysql legacy < mariadb.dump

mysql_secure_installation
	set password - y
	zaldebro
mysql -u root -p
	>use legacy;
	>show tables;
	>select * from TABLE;
	>create user 'raikon'@'localhost' identified by 'zaldebro' ;
	>grant all on legacy.* to raikon@'localhost';
